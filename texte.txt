function setStep(n) {
  document.getElementById('step1').classList.toggle('hidden', n!==1);
  document.getElementById('step2').classList.toggle('hidden', n!==2);
  document.getElementById('step3').classList.toggle('hidden', n!==3);
  for(let i=1;i<=3;i++){
    const d = document.getElementById('dot'+i);
    d.classList.remove('active','done');
    if(i===n) d.classList.add('active');
    else if(i<n) d.classList.add('done');
  }
}

function showSymptoms(data) {
  const container = document.getElementById('symptomTags');
  container.innerHTML = '';
  data.tags.forEach(tag => {
    const el = document.createElement('div');
    el.className = 'sym-tag';
    el.textContent = tag;
    el.onclick = () => toggleSymptom(el, tag);
    container.appendChild(el);
  });
}

function toggleSymptom(el, tag) {
  el.classList.toggle('selected');
  if(selectedSymptoms.includes(tag)) selectedSymptoms = selectedSymptoms.filter(s=>s!==tag);
  else selectedSymptoms.push(tag);
}

function askClaude() {
  const freeText = document.getElementById('freeInput').value.trim();
  const symsText = selectedSymptoms.length ? selectedSymptoms.join(', ') : '';
  const query = freeText || symsText || 'informations générales';

  if(!selectedAge && !selectedCategory && !freeText) {
    alert('Veuillez sélectionner l\'âge ou décrire votre situation.');
    return;
  }

  const ageInfo = selectedAge ? `Enfant de ${selectedAge}` : 'Âge non précisé';
  const catInfo = selectedCategory || '';

  // Chercher d'abord dans les réponses admin
  const adminRep = getAdminReponse(selectedAge, catInfo, freeText, selectedSymptoms);
  const reponse = adminRep || trouverReponse(selectedAge, catInfo, selectedSymptoms, freeText);

  // Afficher la section résultat
  document.getElementById('resultSection').classList.remove('hidden');
  document.getElementById('quickSection').classList.add('hidden');
  document.getElementById('resultIcon').textContent = reponse.icon;
  document.getElementById('resultTitle').textContent = reponse.titre;
  document.getElementById('resultMeta').textContent = `${ageInfo}${catInfo ? ' · ' + catInfo : ''}`;
  document.getElementById('resultBody').innerHTML = reponse.contenu;

  document.getElementById('resultSection').scrollIntoView({ behavior: 'smooth', block: 'start' });
}

function renderResult(text) {
  // Convert markdown-like to HTML
  let html = text
    .replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>')
    .replace(/\*([^*]+)\*/g, '<em>$1</em>')
    .replace(/^### (.+)$/gm, '<h3>$1</h3>')
    .replace(/^## (.+)$/gm, '<h3>$1</h3>')
    .replace(/^# (.+)$/gm, '<h3>$1</h3>')
    .replace(/^\d+\. \*\*([^*]+)\*\*/gm, '<h3>$1</h3>')
    .replace(/^\d+\. (.+)$/gm, '<p><strong>$1</strong></p>')
    .replace(/^[-•] (.+)$/gm, '<li>$1</li>')
    .replace(/(<li>.*<\/li>)/gs, (m) => '<ul>' + m + '</ul>')
    .replace(/\n\n/g, '</p><p>')
    .replace(/\n/g, '<br>');

  // Wrap alerts
  html = html.replace(/(⚠️[^<]+(<br>|<\/p>))/g, '<div class="alert-box urgent"><div class="alert-title">⚠️ Attention</div>$1</div>');
  html = html.replace(/(💡[^<]+(<br>|<\/p>))/g, '<div class="alert-box"><div class="alert-title">💡 Conseil</div>$1</div>');

  document.getElementById('resultBody').innerHTML = html;
}

  document.getElementById('tabAdd').style.background = tab === 'add' ? '#EFA07A' : 'rgba(239,160,122,0.15)';
  document.getElementById('tabAdd').style.color = tab === 'add' ? 'white' : '#8B5E3C';
  if (tab === 'list') renderAdminList();
  if (tab === 'add') {
    editingIndex = -1;
    document.getElementById('adminFormTitle').textContent = '➕ Nouvelle réponse';
    document.getElementById('formTitre').value = '';
    document.getElementById('formIcon').value = '🌸';
    document.getElementById('formAge').value = '0_12';
    document.getElementById('formKeywords').value = '';
    document.getElementById('formExplication').value = '';
    document.getElementById('formConseils').value = '';
    document.getElementById('formAlerte').value = '';
    document.getElementById('formAstuce').value = '';
    document.getElementById('formSuccess').style.display = 'none';
  }
}
function renderAdminList() {
  const list = document.getElementById('adminReponsesList');
  const empty = document.getElementById('emptyAdmin');
  if (adminReponses.length === 0) {
    list.innerHTML = '';
    empty.style.display = 'block';
    return;
  }
  empty.style.display = 'none';
  list.innerHTML = adminReponses.map((r, i) => `
    <div style="background:white;border:1px solid rgba(239,160,122,0.25);border-radius:12px;padding:12px 14px;display:flex;align-items:center;gap:10px;">
      <span style="font-size:1.4rem;">${r.icon}</span>
      <div style="flex:1;">
        <div style="font-size:0.88rem;font-weight:600;color:#5C3D1E;">${r.titre}</div>
        <div style="font-size:0.72rem;color:#C4A896;margin-top:2px;">${r.age === '0_12' ? '0–12 mois' : r.age === '1_3' ? '1–3 ans' : '3–5 ans'} · ${r.keywords.split(',').slice(0,3).join(', ')}…</div>
      </div>
      <button onclick="editReponse(${i})" style="background:rgba(239,160,122,0.15);border:none;border-radius:8px;padding:6px 10px;font-size:0.78rem;cursor:pointer;color:#8B5E3C;">✏️ Modifier</button>
      <button onclick="deleteReponse(${i})" style="background:rgba(220,80,80,0.1);border:none;border-radius:8px;padding:6px 10px;font-size:0.78rem;cursor:pointer;color:#B03030;">🗑️</button>
    </div>
  `).join('');
}
function editReponse(i) {
  editingIndex = i;
  const r = adminReponses[i];
  document.getElementById('adminFormTitle').textContent = '✏️ Modifier la réponse';
  document.getElementById('formTitre').value = r.titre;
  document.getElementById('formIcon').value = r.icon;
  document.getElementById('formAge').value = r.age;
  document.getElementById('formKeywords').value = r.keywords;
  document.getElementById('formExplication').value = r.explication;
  document.getElementById('formConseils').value = r.conseils;
  document.getElementById('formAlerte').value = r.alerte;
  document.getElementById('formAstuce').value = r.astuce;
  document.getElementById('formSuccess').style.display = 'none';
  adminTab('add');
}
function deleteReponse(i) {
  if (confirm('Supprimer cette réponse ?')) {
    adminReponses.splice(i, 1);
    localStorage.setItem('sos_admin_reponses', JSON.stringify(adminReponses));
    renderAdminList();
  }
}
function saveReponse() {
  const titre = document.getElementById('formTitre').value.trim();
  const icon = document.getElementById('formIcon').value.trim() || '🌸';
  const age = document.getElementById('formAge').value;
  const keywords = document.getElementById('formKeywords').value.trim();
  const explication = document.getElementById('formExplication').value.trim();
  const conseils = document.getElementById('formConseils').value.trim();
  const alerte = document.getElementById('formAlerte').value.trim();
  const astuce = document.getElementById('formAstuce').value.trim();
  if (!titre || !explication) { alert('Remplissez au moins le titre et l\'explication.'); return; }
  const obj = { titre, icon, age, keywords, explication, conseils, alerte, astuce };
  if (editingIndex >= 0) adminReponses[editingIndex] = obj;
  else adminReponses.push(obj);
  localStorage.setItem('sos_admin_reponses', JSON.stringify(adminReponses));
  document.getElementById('formSuccess').style.display = 'block';
  setTimeout(() => { adminTab('list'); }, 1500);
}

// Injecter les réponses admin dans la recherche
function getAdminReponse(age, categorie, texteLibre, symptomes) {
  if (adminReponses.length === 0) return null;
  const tout = (symptomes.join(' ') + ' ' + texteLibre + ' ' + categorie + ' ' + age).toLowerCase();
  for (const r of adminReponses) {
    const kws = r.keywords.toLowerCase().split(',').map(k => k.trim()).filter(Boolean);
    const ageMatch = !r.age || r.age === '0_12' && (age.includes('mois')) ||
      r.age === '1_3' && (age === '1-2 ans' || age === '2-3 ans') ||
      r.age === '3_5' && age === '3-5 ans' || !age;
    const kwMatch = kws.some(k => k && tout.includes(k));
    if (kwMatch) {
      const conseilsHtml = r.conseils ? '<ul>' + r.conseils.split('\n').filter(Boolean).map(c => `<li>${c}</li>`).join('') + '</ul>' : '';
      const alerteHtml = r.alerte ? `<div class="alert-box urgent"><div class="alert-title">⚠️ Quand consulter</div><p>${r.alerte}</p></div>` : '';
      const astuceHtml = r.astuce ? `<div class="alert-box"><div class="alert-title">💡 Astuce parents</div><p>${r.astuce}</p></div>` : '';
      return {
        titre: r.titre, icon: r.icon,
        contenu: `<h3>Ce qui se passe</h3><p>${r.explication}</p>${r.conseils ? '<h3>Ce que vous pouvez faire</h3>' + conseilsHtml : ''}${alerteHtml}${astuceHtml}`
      };
    }
  }
  return null;
}
